  def test_class_lookup
    a = Residue::A
  
    assert_equal a, Residue['A']
    assert_equal a, Residue['Ala']
    assert_equal a, Residue['Alanine']

    assert_nil Residue['X']
    assert_nil Residue['BACKBONE']
    assert_raise(NoMethodError) { Residue.backbone }
  end

  def test_parse_simple_documentation
    assert_equal "H(2)O", EmpiricalFormula.parse_simple("H(2)O").to_s
    assert_equal "H(2)O", EmpiricalFormula.parse_simple("H (2) O").to_s
    assert_equal "H(2)O", EmpiricalFormula.parse_simple("HO(-1)O(2)H").to_s
  end

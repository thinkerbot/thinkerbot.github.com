  def test_parse_simple_speed
    benchmark_test(20) do |x|
      n = 10
       
      ["H(20)","H(2)H(2)H(2)H(2)"].each do |formula|
        x.report("#{n}k #{formula}") do 
          (n*1000).times { EmpiricalFormula.parse_simple(formula) }
        end
      end
    end
  end

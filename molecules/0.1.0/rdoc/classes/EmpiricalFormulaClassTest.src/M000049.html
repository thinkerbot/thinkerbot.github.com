  def test_parse_fails_for_malformed_formulae
    [
      # mismatched parenthesis

      "H)2",
      "(H2", 
      "(O2(H2)",
      "(O)2H2)", 
      # hanging factors

      "2C", 
      #"(2)",

      "(2)2", 
      "(2C)", 
      "(2C)2",
      "C(2C)",
      # empty parenthesis

      "()",
      "()2"
    ].each do |formula|
      assert_raise(EmpiricalFormula::ParseError) { EmpiricalFormula.parse(formula) }
    end
  end

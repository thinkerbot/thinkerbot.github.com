  def test_each_residue_speed
    benchmark_test(20) do |x|
      p = Polypeptide.new("RPPGFSPFR" * 10) 
      
      x.report("1k RPPGFSPFR * 10") do 
        1000.times { p.each_residue {|r| r} }
      end
      x.report("1k each_byte:") do 
        1000.times { p.sequence.each_byte {|b| b} }
      end
    end
  end

  def test_each_residue_returns_each_residue_sequentially
    residues = []
    p = Polypeptide.new("\sRP PG\t F")
    p.each_residue {|r| residues << r}
    
    assert_equal [Residue::R, Residue::P, Residue::P, Residue::G, Residue::F], residues
  end

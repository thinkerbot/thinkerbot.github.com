  def test_multiply_speed
    benchmark_test(30) do |x|
      n = 100
      x.report("#{n}k multiply([1,3,-1], 2)") do 
        (n*10**3).times { multiply([1,3,-1], 2) }
      end
      
      x.report("#{n}k multiply([1,3,-1], 0)") do 
        (n*10**3).times { multiply([1,3,-1], 0) }
      end
    end
  end

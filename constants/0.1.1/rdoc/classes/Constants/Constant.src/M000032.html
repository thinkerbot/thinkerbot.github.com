      def parse(str)
        str = str.to_s.gsub(/\s/, '')
        raise "cannot parse: #{str}" unless str =~ /^(-?\d+)(\.\d+)?(\(\d+\))?(e-?\d+)?(.*)$/

        value = "#{$1}#{$2}#{$4}".to_f
        unit = $5
        uncertainty = case
        when $3 == nil then nil
        else
          factor = $2 == nil ? 0 : 1 - $2.length
          factor += $4[1..-1].to_i unless $4 == nil
          $3[1..-2].to_i * 10 ** factor
        end

        block_given? ? yield(value, unit, uncertainty) : new(value, unit, uncertainty)
      end

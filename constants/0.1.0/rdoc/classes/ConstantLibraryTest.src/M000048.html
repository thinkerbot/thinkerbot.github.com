  def test_clear_clears_all_values_from_lib_indicies_and_collections
    lib.index_by("str") {|v| v.to_s }
    lib.collect("str") {|v| v.to_s }
    
    assert !lib.values.empty?
    assert !lib.indicies['str'].empty?
    assert !lib.collections['str'].empty?
    
    lib.clear
    
    assert lib.values.empty?
    assert lib.indicies['str'].empty?
    assert lib.collections['str'].empty?
  end

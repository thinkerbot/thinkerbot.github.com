  def test_clear_only_removes_indicies_and_collections_if_specified
    lib.index_by("str") {|v| v.to_s }
    lib.collect("str") {|v| v.to_s }
    
    lib.clear
    
    assert !lib.indicies.empty?
    assert !lib.collections.empty?
    
    lib.clear(true)
    
    assert lib.indicies.empty?
    assert lib.collections.empty?
  end

  def test_collect_uses_name_as_collections_key
    lib.collect(:sym) {|v| v.to_s }
    assert lib.collections.has_key?(:sym)
    assert !lib.collections.has_key?('sym')
    
    lib.collect('str') {|v| v.to_s }
    assert !lib.collections.has_key?(:str)
    assert lib.collections.has_key?('str')
  end

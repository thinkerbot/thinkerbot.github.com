set(options) {|| ...}

Set or unset options as specified. For example: 

  set 'x' => true, 'v' => false

If a block is given then options will only be reset when the block completes. [Spec]

[Source]

     # File lib/linebook/os/posix/utilities.rb, line 542
542:         def set(options)
543:           if block_given?
544:             var = _package_.next_variable_name('set')
545:             patterns = options.keys.collect {|key| "-e #{key}" }.sort
546:             writeln %{#{var}=$(set +o | grep #{patterns.join(' ')})}
547:           end
548:         
549:           options.keys.sort_by {|opt| opt.to_s }.each do |opt|
550:             writeln %{set #{options[opt] ? '-' : '+'}o #{opt}}
551:           end
552:         
553:           if block_given?
554:             yield
555:             writeln %{eval "$#{var}"}
556:           end
557:           chain_proxy
558:         end
